Two way binding:-
Two-way binding connects data (state) and view (UI) so changes in one are instantly reflected in the other.
Like this:

import { useState } from "react";

function App() {
  const [name, setName] = useState("");

  return (
    <>
      <input
        value={name}
        onChange={(e) => setName(e.target.value)}
      />
      <p>Hello {name}</p>
    </>
  );
}
